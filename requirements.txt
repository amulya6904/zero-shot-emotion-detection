# INSTALLATION NOTE - GPU vs CPU
# The PyPI `torch`, `torchvision`, and `torchaudio` wheels may install CPU-only
# builds by default depending on your platform and pip configuration. For GPU
# builds, install the PyTorch packages from the official PyTorch index that
# matches your CUDA runtime, then install the rest of this file.
#
# Example CPU install:
#   pip install -r requirements.txt
#
# Example CUDA 11.8 install:
#   pip install torch==2.1.0 torchvision==0.16.0 torchaudio==2.1.0 \
#       --index-url https://download.pytorch.org/whl/cu118
#   pip install -r requirements.txt --no-deps
#
# DEVELOPMENT INSTALL
# Create and activate a virtual environment before installing:
#   python -m venv .venv
#   .venv\Scripts\activate
#   python -m pip install --upgrade pip
#   pip install -r requirements.txt
#
# CUDA COMPATIBILITY NOTE

# your NVIDIA driver supports the CUDA build you choose. Mixing a CUDA-enabled
# PyTorch wheel with an unsupported local driver version will cause runtime
# failures. If you do not need GPU acceleration, prefer the CPU build.
